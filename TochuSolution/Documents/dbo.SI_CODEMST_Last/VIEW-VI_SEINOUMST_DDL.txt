CREATE VIEW VI_SEINOUMST AS
SELECT
    PRODUCT,
    NOUSCD,
    MAX(CASE WHEN NUM = 10 THEN MIN END) AS T10A,
MAX(CASE WHEN NUM = 10 THEN MAX END) AS T10B,
MAX(CASE WHEN NUM = 20 THEN MIN END) AS T20A,
MAX(CASE WHEN NUM = 20 THEN MAX END) AS T20B,
MAX(CASE WHEN NUM = 30 THEN MIN END) AS T30A,
MAX(CASE WHEN NUM = 30 THEN MAX END) AS T30B,
MAX(CASE WHEN NUM = 40 THEN MIN END) AS T40A,
MAX(CASE WHEN NUM = 40 THEN MAX END) AS T40B,
MAX(CASE WHEN NUM = 50 THEN MIN END) AS T50A,
MAX(CASE WHEN NUM = 50 THEN MAX END) AS T50B,
MAX(CASE WHEN NUM = 60 THEN MIN END) AS T60A,
MAX(CASE WHEN NUM = 60 THEN MAX END) AS T60B,
MAX(CASE WHEN NUM = 70 THEN MIN END) AS T70A,
MAX(CASE WHEN NUM = 70 THEN MAX END) AS T70B,
MAX(CASE WHEN NUM = 80 THEN MIN END) AS T80A,
MAX(CASE WHEN NUM = 80 THEN MAX END) AS T80B,
MAX(CASE WHEN NUM = 90 THEN MIN END) AS T90A,
MAX(CASE WHEN NUM = 90 THEN MAX END) AS T90B,
MAX(CASE WHEN NUM = 100 THEN MIN END) AS T100A,
MAX(CASE WHEN NUM = 100 THEN MAX END) AS T100B,
MAX(CASE WHEN NUM = 110 THEN MIN END) AS T110A,
MAX(CASE WHEN NUM = 110 THEN MAX END) AS T110B,
MAX(CASE WHEN NUM = 120 THEN MIN END) AS T120A,
MAX(CASE WHEN NUM = 120 THEN MAX END) AS T120B,
MAX(CASE WHEN NUM = 130 THEN MIN END) AS T130A,
MAX(CASE WHEN NUM = 130 THEN MAX END) AS T130B,
MAX(CASE WHEN NUM = 140 THEN MIN END) AS T140A,
MAX(CASE WHEN NUM = 140 THEN MAX END) AS T140B,
MAX(CASE WHEN NUM = 150 THEN MIN END) AS T150A,
MAX(CASE WHEN NUM = 150 THEN MAX END) AS T150B,
MAX(CASE WHEN NUM = 160 THEN MIN END) AS T160A,
MAX(CASE WHEN NUM = 160 THEN MAX END) AS T160B
    FROM
    SI_SEINOUMST
GROUP BY
    PRODUCT,
    NOUSCD
